indicator("MA", overlay=true)

// Get user input
maLength = input.int(title="MA Length", defval=50, minval=1)

// Get MA
ema = ta.ema(close, maLength)

// Get MA Filter
maFilter = close > ema ? true : false

// Draw MA
plot(ema, title="EMA", color=maFilter ? color.green : color.red, linewidth=2)
